CREATE OR REPLACE PROCEDURE assign_for_task(IN task_id INT, IN volunteer_ids INT[], IN hours_limit INT DEFAULT 8)
AS 
$body$
DECLARE 
	report TEXT;
    task_state TEXT;
	examined_volunteer INT;
    volunteer_competency_set TABLE (volunteer_id INT, competency_id INT);
	cu_volunteer_competency REFCURSOR; 
	volunteer_competency_pair RECORD;
BEGIN
    SELECT s.name FROM task t JOIN status s ON t.status_id = s.id 
    WHERE t.id = $1 INTO task_state;

    IF task_state NOT IN ('Planned', 'Cancelled') THEN
        RAISE EXCEPTION 'Task already %', task_state;
    ELSE
		report := report || 'Assigment report\n';
		INSERT INTO competency_volunteer_set (volunteer_id, competency_id)
		SELECT u.volunteer_id, tc.competency_id
		FROM task_competency tc
		CROSS JOIN unnest(volunteer_ids) AS u(volunteer_id)
		WHERE tc.task_id = $1;

		FOR EACH examined_volunteer IN $2
		LOOP
			OPEN cu_volunteer_competency FORWARD ONLY FOR
			volunteer_competency_set WHERE volunteer_id = examined_volunteer.id;
			LOOP
				FETCH cu_volunteer_competency INTO volunteer_competency_pair;
				EXIT WHEN NOT FOUND;
				IF EXISTS (
					SELECT 1
					FROM volunteer_competency vc 
					WHERE vc.volunteer_id = volunteer_competency_pair.volunteer_id
					AND vc.competency_id = volunteer_competency_pair.competency_id 
					) AND get_hours(volunteer_competency_pair.volunteer_id) < hours_limit
				THEN
					IF NOT EXISTS (
							SELECT 1
							FROM volunteer_task vt
							WHERE vt.volunteer_id = volunteer_competency_pair.volunteer_id
							AND vt.task_id = $1
						)
						THEN
							INSERT INTO volunteer_task (volunteer_id, task_id) 
							VALUES (volunteer_competency_pair.volunteer_id, $1);
					END IF;
				ELSE
				report := report || 'Volunteer '||volunteer_competency_pair.volunteer_id
				||'does not have competency:'||volunteer_competency_pair.competency_id||'\n';
				END IF;
				END IF;
			END LOOP;
		END LOOP;	
    END IF;
	IF (SELECT COUNT(*) FROM volunteer_task vt
	   WHERE vt.task_id = $1) = 0
	   OR
	   (SELECT COUNT(*) FROM volunteer_task vt
	   WHERE vt.task_id = $1) <> array_length(volunteer_ids, 1);
	THEN
	RAISE EXCEPTION 'Assigment failed. See report:\n'
	ELSE UPDATE task SET status_id = 2 WHERE task.id = $1;
	END IF;
END
$body$
LANGUAGE plpgsql;
